/* ── DATE HELPERS ── */
function fDate(ts) {
  const d = new Date(ts);
  return `${d.getFullYear()}.${pad(d.getMonth()+1)}.${pad(d.getDate())}`;
}
function fTime(ts) {
  const d = new Date(ts);
  const h = d.getHours(), m = pad(d.getMinutes());
  return `${h<12?'오전':'오후'} ${h%12||12}:${m}`;
}
function fLabel(ts) {
  const now = new Date();
  const y = new Date(now); y.setDate(now.getDate()-1);
  if (fDate(ts)===fDate(now)) return '오늘';
  if (fDate(ts)===fDate(y))   return '어제';
  const t = new Date(ts);
  return `${t.getMonth()+1}월 ${t.getDate()}일`;
}
function tAgo(ts) {
  const d = Date.now()-ts;
  if (d<60000) return '방금';
  if (d<3600000) return `${Math.floor(d/60000)}분 전`;
  if (d<86400000) return `${Math.floor(d/3600000)}시간 전`;
  return `${Math.floor(d/86400000)}일 전`;
}
const pad = n => String(n).padStart(2,'0');

function getList() {
  const q = (document.getElementById('sq')?.value||'').toLowerCase();
  return DB.filter(i=>{
    if (tab!=='all' && i.cat!==tab) return false;
    if (q && !i.name.toLowerCase().includes(q)) return false;
    if (chip==='low' && !(i.qty>0 && i.qty<=(i.min||0))) return false;
    if (chip==='out' && i.qty>0) return false;
    return true;
  }).sort((a,b)=>{
    if (order==='name')     return a.name.localeCompare(b.name,'ko');
    if (order==='qty_asc')  return a.qty-b.qty;
    if (order==='qty_desc') return b.qty-a.qty;
    if (order==='updated')  return (b.upd||0)-(a.upd||0);
    return 0;
  });
}

/* ── ADJUST QTY ── */
function adj(id, delta) {
  const item = DB.find(i=>i.id===id);
  if (!item) return;
  const prev = item.qty;
  item.qty = Math.max(0, item.qty+delta);
  if (item.qty!==prev) {
    if (!item.hist) item.hist=[];
    item.hist.unshift({d:delta, r:item.qty, ts:Date.now(), m:''});
    if (item.hist.length>200) item.hist.pop();
    item.upd = Date.now();
    persist(); render();
    if (item.qty===0) toast(`⚠️ ${item.name} 품절!`);
    else if (item.min && item.qty===item.min) toast(`⚠️ ${item.name} 재고 부족`);
  }
}

function fillSelect(selId, customId, items, val) {
  const sel = document.getElementById(selId);
  const custom = document.getElementById(customId);
  sel.innerHTML = items.map(function(v) {
    return '<option value="' + v + '">' + v + '</option>';
  }).join('') + '<option value="__other__">직접 입력...</option>';

  if (val && items.indexOf(val) === -1 && val !== '') {
    // 기존 값이 목록에 없으면 직접입력 표시
    sel.value = '__other__';
    custom.style.display = 'block';
    custom.value = val;
  } else {
    sel.value = val || items[0] || '';
    custom.style.display = 'none';
    custom.value = '';
  }
}

function handleSelOther(sel, customId) {
